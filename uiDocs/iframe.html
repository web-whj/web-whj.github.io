    # Iframe 页面

    <j-iframe src="/" style="height:430px" v-on:loaded="iFrameLoaded"></j-iframe>
<script>
  export default {
    data() {
      return {
        
      }
    },
    methods: {
      iFrameLoaded(){
        this.$message({
          message: '加载成功！',
          type: 'success'
        });
      }
    }
  };
</script>
